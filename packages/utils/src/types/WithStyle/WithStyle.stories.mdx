# WithStyle

The `WithStyle` function helps add styled component class reference as a property to the component. 
You can then use `MyComponent.Style` to override styles of `MyComponent` in any other style component.

### Example

```ts
import { WithStyle } from '@medly-components/utils';
import { memo } from 'react';
import { Wrapper } from './Content.styled';
import type { FC } from 'react';

const Component: FC = memo(props => {
    return <Wrapper {...props} />;
});
Component.displayName = 'Content';
export const Content: FC & WithStyle = Object.assign(Component, { Style: Wrapper });
```
